    <script>
        function sq(x) { return x * x }

        const canvas = document.getElementById('gameCanvas');
        const ctx = canvas.getContext('2d');
        let nearChest;

        const gridSize = 120; // 720 / 6 = 120px per square

        // Draw grid ONCE on background
        function drawGridBackground() {
            ctx.strokeStyle = '#fff';
            ctx.lineWidth = 1;

            for (let x = 0; x <= canvas.width; x += gridSize) {
                ctx.beginPath();
                ctx.moveTo(x, 0);
                ctx.lineTo(x, canvas.height);
                ctx.stroke();
            }

            for (let y = 0; y <= canvas.height; y += gridSize) {
                ctx.beginPath();
                ctx.moveTo(0, y);
                ctx.lineTo(canvas.width, y);
                ctx.stroke();
            }
        }

        const player = {
            x: 20,
            y: 20,
            width: 60,
            height: 60,
            // width: 60,
            // height: 107,
            speed: 5,
            sprite: true,
        };
        const obstacles = [
            { x: 200, y: 300, height: 50, width: 75, useImage: true, source: "assets/brick.png" }
        ];
        const chests = [
            { x: 400, y: 200, height: 50, width: 50, useImage: true, source: "assets/chest.png" }
        ]
        obstacles.forEach(obstacle => {
            if (obstacle.useImage && obstacle.source) {
                obstacle.image = new Image();
                obstacle.image.src = obstacle.source;
            }
        });

        chests.forEach(chest => {
            if (chest.useImage && chest.source) {
                chest.image = new Image();
                chest.image.src = chest.source;
            }
        });
        const keys = {};

        window.addEventListener('keydown', function (e) {
            keys[e.key] = true;

            if (e.key === 'r' && nearChest) {
                const inv = document.getElementById('inventory');
                inv.classList.toggle('none');
            }

            if (e.key === 'Escape') {
                document.getElementById('inventory').classList.add('none');
            }
        });


        window.addEventListener('keyup', function (e) {
            keys[e.key] = false;
        });

        function rectsOverlap(r1, r2) {
            return !(r1.x + r1.width <= r2.x ||
                r1.x >= r2.x + r2.width ||
                r1.y + r1.height <= r2.y ||
                r1.y >= r2.y + r2.height);
        }

        function update() {
            let prevX = player.x;
            let prevY = player.y;

            if (keys['w'] || keys['ArrowUp']) player.y -= player.speed;
            if (keys['a'] || keys['ArrowLeft']) player.x -= player.speed;
            if (keys['s'] || keys['ArrowDown']) player.y += player.speed;
            if (keys['d'] || keys['ArrowRight']) player.x += player.speed;

            if (keys['Shift']) {
                player.speed = 8;
            } else {
                player.speed = 5;
            }

            // Keep player inside canvas bounds
            if (player.x < 0) player.x = 0;
            if (player.y < 0) player.y = 0;
            if (player.x + player.width > canvas.width) player.x = canvas.width - player.width;
            if (player.y + player.height > canvas.height) player.y = canvas.height - player.height;

            // Check collisions against obstacles
            for (const obs of obstacles) {
                if (rectsOverlap(player, obs)) {
                    // Undo move on collision
                    player.x = prevX;
                    player.y = prevY;
                    break;
                }
            }
            for (const chest of chests) {
                if (rectsOverlap(player, chest)) {
                    // Undo move on collision
                    player.x = prevX;
                    player.y = prevY;
                    break;
                }
            }



            document.querySelector(".cords").innerHTML = `(${player.x},${player.y})`;
            let setting = localStorage.getItem("cordsdisplaysetting");
            document.querySelector(".cords").style.display = setting === "Disable" ? "none" : "flex";
            let wasNearChest = nearChest; // track previous state

            nearChest = false; // default to false, check below to set true

            chests.forEach(chest => {
                const dx = (chest.x + chest.width / 2) - (player.x + player.width / 2);
                const dy = (chest.y + chest.height / 2) - (player.y + player.height / 2);
                const distance = Math.sqrt(dx * dx + dy * dy);

                if (distance <= 100) {
                    nearChest = true;
                    if (!wasNearChest) {
                        console.log("CHEST ENEMY DETECTED");
                        document.querySelector(".interaction").classList.remove("none");
                    }
                }
            });

            if (!nearChest && wasNearChest) {
                document.querySelector(".interaction").classList.add("none");
            }


        }







        const obstacleImg = new Image();
        obstacleImg.src = 'assets/brick.png';  // path to your image file

        const sprite = new Image()
        sprite.src = 'assets/sprite.png';

        function draw() {
            ctx.clearRect(0, 0, canvas.width, canvas.height);

            drawGridBackground();

            obstacles.forEach(obstacle => {
                if (obstacle.useImage && obstacle.image && obstacle.image.complete) {
                    ctx.drawImage(obstacle.image, obstacle.x, obstacle.y, obstacle.width, obstacle.height);
                } else {
                    ctx.fillStyle = "brown";
                    ctx.fillRect(obstacle.x, obstacle.y, obstacle.width, obstacle.height);
                }
            });
            chests.forEach(chest => {
                if (chest.useImage && chest.image && chest.image.complete) {
                    ctx.drawImage(chest.image, chest.x, chest.y, chest.width, chest.height);
                } else {
                    ctx.fillStyle = "brown";
                    ctx.fillRect(chest.x, chest.y, chest.width, chest.height);
                }
            });
            if (player.sprite && sprite.complete) {
                ctx.drawImage(sprite, player.x, player.y, player.width, player.height);
            } else {
                ctx.fillStyle = 'red';
                ctx.fillRect(player.x, player.y, player.width, player.height);
            }




        }


        function gameLoop() {
            update();
            draw();
            requestAnimationFrame(gameLoop);
        }

        gameLoop();
    </script>
